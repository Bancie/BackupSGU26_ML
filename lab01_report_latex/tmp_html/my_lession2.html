def NhapDuLieu():
    """
    Input:
        Ban phim = 1 5 6
    Output:
        (a, b, c) --> (1, 5, 6)
    """
    while True:
        line = input("Moi ban nhap he so a, b, c: ")
        parts = line.split()
        if len(parts) != 3:
            print("Ban nhap sai! Moi ban nhap lai!")
            continue
        try:
            a, b, c = float(parts[0]), float(parts[1]), float(parts[2])
        except ValueError:
            print("Ban nhap sai! Moi ban nhap lai!")
            continue
        return a, b, c

def GiaiPhuongTrinhBac2(a, b, c):
    """
    Input: a, b, c
    Output:
    + flag = -1 (VSN), 0 (VN), k (k nghiem)
    + () --> flag = -1, 0
    + (x) --> flag = 1
    + (x1, x2) --> flag = 2
    """
    import math

    if a == 0:
        if b == 0 and c == 0:
            return -1, ()
        if b == 0 and c != 0:
            return 0, ()
        return 1, (-c / b,)

    delta = b * b - 4 * a * c
    if delta < 0:
        return 0, ()
    if delta == 0:
        x = -b / (2 * a)
        return 1, (x,)
    sqrt_d = math.sqrt(delta)
    x1 = (-b - sqrt_d) / (2 * a)
    x2 = (-b + sqrt_d) / (2 * a)
    return 2, (x1, x2)

import random


def SinhNgauNhien(n, vmin=-10, vmax=10):
    """
    Input:
        n = 30, vmin = -10, vmax = 10
    Output:
        a = [-1, -4, -8, 6, 4, -5, -1, -4, -9, -5, -10, 0, -6, 2, 1, -9, -9, 10, -8, -9, -7, 9, 2, -4, -1, -6, 5, -6, 9, -8]
    """
    a = []
    for _ in range(n):
        a.append(random.randint(vmin, vmax))
    return a

def DemTongChanLe(a):
    """
    Input: a[] = [1, 4, 2, 0, -1, -7, 3, -6, 4, 9, -5, 7, 7, -7, -10, 1, -10, 9, -4, -6, 10, -2, -10, -8, -3, 2, -7, -10, 6, 0]
    Output: tong, sochan, sole
    """
    tong = 0
    sochan = 0
    sole = 0
    for v in a:
        tong += v
        if v % 2 == 0:
            sochan += 1
        else:
            sole += 1
    return tong, sochan, sole

def DemSoLanXuatHien(a):
    """
    Input:
        a = ['A', 'G', 'C', 'A', 'C', 'F', 'F', 'T', 'H']
    Output:
        dem = {'A': 2, 'G': 1, 'C': 2, 'F': 2, 'T': 1, 'H': 1}
    """
    dem = {}
    for ch in a:
        dem[ch] = dem.get(ch, 0) + 1
    return dem
